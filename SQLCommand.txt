- Cоздание таблицы Posts(таблицы с обьявлениями пользователей)

CREATE TABLE posts
(
    id INT AUTO_INCREMENT PRIMARY KEY,
    login VARCHAR(30) NOT NULL,
    location VARCHAR(50) NOT NULL,
    price INT NOT NULL,
    description VARCHAR(200)
);
